# 创建工作目录
mkdir -p ~/wechat && cd ~/wechat
# 生成docker-compose.yml配置文件
cat > docker-compose.yml << 'EOF'
version: '3'
services:
  wechat-selkies:
    image: nickrunning/wechat-selkies:latest
    container_name: wechat-selkies
    ports:
      - "8080:8080"  # 浏览器访问端口，可自定义（如8081）
    environment:
      - password=Wzx12345@  # 必须设置！防止他人访问，建议8位以上字母+数字
      - geometry=1920x1080  # 微信界面分辨率，可按需调整
      - timezone=Asia/Shanghai  # 时区设置，保证时间正确
    volumes:
      - ./wechat-data:/data  # 数据持久化，保存聊天记录
      - /dev/shm:/dev/shm    # 提升运行性能
    restart: unless-stopped  # 服务器重启自动恢复
EOF
